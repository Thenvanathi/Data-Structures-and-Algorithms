// Perfect Number 

class Main{
    public static void main(String[] args){
        int n = 28;
        int sum = 1;
        
        for(int i=2; i<=Math.sqrt(n); i++){
            if(n % i == 0){
                sum = sum + i;
                if(n/i != i){
                    sum = sum + n/i;
                }
            }
        }
        if( n == sum){
            System.out.println("Perfect Number");
        } else {
            System.out.println("Not Perfect Number");
        }
    }
}
