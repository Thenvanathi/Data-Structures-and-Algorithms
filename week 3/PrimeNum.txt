// Prime Number 

class Main{
    public static void main(String[] args){
        int n = 12; 
        int flat = 1;
        
        for(int i=2; i<=Math.sqrt(n); i++){
            if(n%i == 0){
                flat = 0;
            }
        }
        if (flat == 1){
            System.out.println("Prime Number");
        } else {
            System.out.println("Not Prime Number");
        }
    }
}
